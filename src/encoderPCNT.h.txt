#ifndef ENCODERPCNT_H
#define ENCODERPCNT_H

#include <Arduino.h>
#include "driver/pcnt.h"

#ifdef __cplusplus
extern "C" {
#endif

typedef struct {
    int pinA;
    int pinB;
    pcnt_unit_t unit;
    volatile int32_t total_count;
    int32_t prevCount; // สำหรับใช้ใน GetDelta
    int16_t last_hw_count;
} Encoder_t;

void Encoder_Init(Encoder_t *enc, int a, int b, pcnt_unit_t unit);
int32_t Encoder_GetCount(Encoder_t *enc);
int32_t Encoder_GetDelta(Encoder_t *enc);
void Encoder_Reset(Encoder_t *enc);

#ifdef __cplusplus
}
#endif

#endif
